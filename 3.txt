#include <iostream>
using namespace std;

class Point {
protected:
    int x, y;

public:
    Point(int x = 0, int y = 0) : x(x), y(y) {}

    Point(const Point& other) {
        x = other.x;
        y = other.y;
    }

    Point& operator=(const Point& other) {
        if (this != &other) {
            x = other.x;
            y = other.y;
        }
        return *this;
    }

    friend istream& operator>>(istream& in, Point& p) {
        in >> p.x >> p.y;
        return in;
    }

    friend ostream& operator<<(ostream& out, const Point& p) {
        out << "(" << p.x << ", " << p.y << ")";
        return out;
    }
};

class ColoredPoint : public Point {
private:
    string color;

public:
    ColoredPoint(int x = 0, int y = 0, string c = "black")
        : Point(x, y), color(c) {
    }

    ColoredPoint(const ColoredPoint& other)
        : Point(other) {
        color = other.color;
    }

    ColoredPoint& operator=(const ColoredPoint& other) {
        if (this != &other) {
            Point::operator=(other);
            color = other.color;
        }
        return *this;
    }

    friend istream& operator>>(istream& in, ColoredPoint& p) {
        in >> static_cast<Point&>(p) >> p.color;
        return in;
    }

    friend ostream& operator<<(ostream& out, const ColoredPoint& p) {
        out << static_cast<const Point&>(p)
            << " color: " << p.color;
        return out;
    }
};

int main() {
    ColoredPoint p1;
    cout << "Enter point (x y color): ";
    cin >> p1;

    ColoredPoint p2 = p1; 
    ColoredPoint p3;
    p3 = p1; 

    cout << "p1: " << p1 << endl;
    cout << "p2 (copy): " << p2 << endl;
    cout << "p3 (assigned): " << p3 << endl;

    return 0;
}
